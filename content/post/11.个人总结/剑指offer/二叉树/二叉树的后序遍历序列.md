输入一个整数数组，判断该数组是不是某二叉搜索树的后序遍历的结果。

如果是则返回true，否则返回false。

假设输入的数组的任意两个数字都互不相同。

#### 样例

```
输入：[4, 8, 6, 12, 16, 14, 10]

输出：true
```





## 解题代码

```cpp
class Solution {
public:
    bool verifySequenceOfBST(vector<int> sequence) {
        //根>左边
        //右边大于根
        return dfs(sequence,0,sequence.size() -1 );
    }
    bool dfs(vector<int> &s ,int l,int r) {
        if(l>=r) return true;
        int root = s[r];
        //左右根
        //找到最后 一个 左子树的序列 满足 sleft < sroot
        int k=0;
        for(int i=0;i<r;++i)
            if(s[i] < root) k = i;
            else break;
        
        //s[k] < root,
        //s[k+1]>= root
        for(int t=k+1;t<=r;++t) {
            if(s[t] < root) {
                return false;
            }
        }
        
        return dfs(s,l,k) && dfs(s,k+1,r-1);
        
    }
    
    
    
};
```
