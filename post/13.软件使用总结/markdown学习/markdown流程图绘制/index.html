; (function () {

  function fmtInt(val) {

    if(val==0) {

      return '0'+val;

    }

    if(val>=1&& val <=9) {

      return '0' + val;

    }

    return ''+val;

  }

  function getFileName() {

    let $ = window.jQuery;

    var node = $('.file-library-file-node.active').eq(0);

    if (!node) {

      return "新内容"

    }

    let path = node.data('path')

    if(!path) {

      return '新内容'

    }

    path = path.replaceAll('\\','/').replaceAll('.md','')

    if(path.indexOf('/') == -1) {

      return path;

    }

    //

    let idx = path.lastIndexOf('/')+1;

    return path.substring(idx);

  

  }

  function getPath() {

    let $ = window.jQuery;

    var node = $('.file-library-file-node.active').eq(0);

    if (!node) {

      return "新内容"

    }

    let path = node.data('path')

    return path;

  }

  function getTypes() {

    let ph = getPath();

    if (!ph) {

      return '';

    }

    var path = ph?.replaceAll('\\','/')?.replaceAll('.md','');

    if(path.indexOf('/') == -1) {

      return path;

    }

    var fileName = getFileName();

    path = path?.replaceAll('/'+fileName,'')

    if(path.indexOf('/') == -1) {

      return path

    }

    return path.substring(path.lastIndexOf('/')+1)

  }

  //添加插件

  window.hugoPlugin = window.hugoPlugin||{};

  window.hugoPlugin.getTypes = getTypes;

  window.hugoPlugin.getFileName = getFileName;

  window.hugoPlugin.getPath = getPath;

  

  window.addEventListener('load', function () {

    let $ = window.jQuery;

    if (!$) {

      window.alert('无法加载 jquery插件')

      return;

    }

    function hugoNew() {

      let date = new Date();

      let m = fmtInt(date.getMonth() + 1);

      let d = fmtInt(date.getDate());

      let y = date.getFullYear();

      let hours = fmtInt( date.getHours());

      let minutes = fmtInt(date.getMinutes());

      let title = getFileName();

      let type = getTypes();

      $.ajax({

        url: 'https://api.mtyqx.cn/api/random.php?return=json',

        timeout : 3500,

        type:'get',

        dataType:'json',

        success: function (res) {

          if (res && res.code && res.imgurl) {

            let url = res.imgurl;

            editor.insertText(

              `---

title: "${title}"

date: ${y}-${m}-${d}T${hours}:${minutes}:48+08:00

lastmod: ${y}-${m}-${d}T${hours}:${minutes}:48+08:00

categories: ["${type}"]

tags: ["${type}"]

author: "lyr"

draft: false

images: ["${url}"]

---

## ${title}

  

    `);

  

          } else {

            console.error('get api url =>', res)

          }

        },

        error: function (e) {

          console.error(e);

  

          editor.insertText(

            `---

title: "${title}"

date: ${y}-${m}-${d}T${hours}:${minutes}:48+08:00

lastmod: ${y}-${m}-${d}T${hours}:${minutes}:48+08:00

categories: ["${type}"]

tags: ["${type}"]

author: "lyr"

draft: false

images: ["https://img.xjh.me/random_img.php"]

---

## ${title}

  

`);

          alert('请求超时');

  
  

        }

      })

    }

  

    window.addEventListener('keyup',function(e) {

      // ctrl + space

      if(e.ctrlKey  && e.which== 32) {

        // console.log('6666')

        hugoNew();

      }

    })

    console.log('ctrl + 空格 ，创建hugo模板');
 
  

  });

  
  
  
  

})();
