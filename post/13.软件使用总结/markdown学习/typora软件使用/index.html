  function fmtInt(val) {
    if(val==0) {
      return '0'+val;
    }
    if(val>=1&& val <=9) {
      return '0' + val;
    }
    return ''+val;
  }
  function getFileName() {
    let $ = window.jQuery;
    var node = $('.file-library-file-node.active').eq(0);
    if (!node) {
      return "新内容"
    }
    let path = node.data('path')
    if(!path) {
      return '新内容'
    }
    path = path.replaceAll('\\','/').replaceAll('.md','')
    if(path.indexOf('/') == -1) {
      return path;
    }
    //
    let idx = path.lastIndexOf('/')+1;
    return path.substring(idx);

  }
  function getPath() {
    let $ = window.jQuery;
    var node = $('.file-library-file-node.active').eq(0);
    if (!node) {
      return "新内容"
    }
    let path = node.data('path')
    return path;
  }
  function getTypes() {
    let ph = getPath();
    if (!ph) {
      return '';
    }
    var path = ph?.replaceAll('\\','/')?.replaceAll('.md','');
    
    if(path.indexOf('/') == -1) {
      return path;
    }
    var fileName = getFileName();
    path = path?.replaceAll('/'+fileName,'')
    if(path.indexOf('/') == -1) {
      return path
    }
    
    return path.substring(path.lastIndexOf('/')+1)
  }

(function() {
  function _renderText()   {
    let $ = window.jQuery;
    //跳过已经被渲染的，防止二次渲染
    let $p = $('#write p,#write p > span:not(.hugo_shortcode)');
    $p.each(function() {
      let $this = $(this);

      $this.contents().filter(function() {
        return this.nodeType === 3;
       }).each(function() {  
         //each text node
        let $this = $(this);
        let prevText = $this.text();
        let currentText = prevText.replace(/{{(.*)(mermaid)(.*)}}/g, function(fullmatch,$1,$2,$3) {
          
          return `<span class="hugo_shortcode shortcode_mermaid">${fullmatch}</span>`
        });
        // console.log(prevText);
        if(prevText != currentText) {
          $this.replaceWith(currentText);
        }
      });
    });


    
  }
    /*
  * fn [function] 需要防抖的函数
  * delay [number] 毫秒，防抖期限值
  */
  function debounce(fn,delay){
    let timer = null //借助闭包
    return function() {
        if(timer){
            clearTimeout(timer) //进入该分支语句，说明当前正在一个计时过程中，并且又触发了相同事件。所以要取消当前的计时，重新开始计时
            timer = setTimeout(fn,delay) 
        }else{
            timer = setTimeout(fn,delay) // 进入该分支说明当前并没有在计时，那么就开始一个计时
        }
    }
  }
  let renderText = debounce(_renderText,6000);
  window.addEventListener('load',function(e) {
    // setTimeout(()=> {
    //   _renderText();
    // },3000);

    const $write = $('#write');
    //监听 元素变化
    // $write.on('DOMSubtreeModified',function(e) {
    //   console.log('change -- node --');
    // })
    // 监听元素变化
    const observer = new MutationObserver((mutations, observer) => {
      console.log(mutations, observer);
      renderText();
    });
    
    observer.observe($write[0], {
      // subtree: true,
      childList:true,
      // attributes: true
    });
    
  });

})();
