public interface IntConsumer{
    void accept(int value);
}
public void repeat (int n,IntConsumer action){
    for (int i=0;i<n;i++){
        action.accept(i);
    }
}
@Test
public void addString(){
    repeat(2,(i)->{
        System.out.println(i);
    });
}
