#define Node ListNode
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if(head==NULL || head->next == NULL) return head;
        //头插法
        //ListNode fake(-1);
        ListNode * p = NULL;
        
        while(true) {
            if(head==NULL) break;
            auto next= head->next;
            head->next = p;
            p = head;
            head = next;
            //head->next = 
            
        }
        
        return p;
        
        
    }
};
