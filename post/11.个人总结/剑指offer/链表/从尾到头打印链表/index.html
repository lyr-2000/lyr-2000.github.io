从尾到头打印链表

返回的结果用数组存储。

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> printListReversingly(ListNode* head) {
        vector<int> res;
        while(true) {
            if(head== NULL) break;
            int curValue = head->val;
            res.push_back(curValue);
            head = head->next;
        }
        reverse(res.begin(),res.end());
        return res;
    }
};
