服务端开发流程

  1. 创建 socket，返回 文件描述符 lfd –socket()
  2. 将 lfd 和 ip port 绑定 –bind()
  3. 将 lfd 由主动监听边被动监听
  4. 接受一个新的连接，得到一个 文件描述符 cfd –accept()
    1. 这个是通信文件描述符 ，用来通信
  5. while(1) { read; send }
  6. 关闭文件描述符 close(lfd) , close(cfd)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/types.h>
 
#include <unistd.h>
#include <arpa/inet.h>
#include <netinet/in.h>


 
int main(void) {
    
    
    int lfd = socket(AF_INET, SOCK_STREAM, 0);
    if(lfd <=0 ) {
        perror("socket error");
        exit(1);
    }
    struct sockaddr_in serv;
    bzero(&serv, sizeof (serv));
    serv.sin_family = AF_INET;
    serv.sin_port = htons(8888);
    serv.sin_addr.s_addr = htonl(INADDR_ANY); //使用本机任意可用IP

    //绑定IP 端口
    int ret = bind(lfd,(struct sockaddr *)&serv,sizeof serv);
    if(ret <0) {
        perror(" error ");
        return -1;
    }
    listen(lfd,128);
    //最大监听 128
    int cfd = accept(lfd, NULL, NULL);
    printf("lfd = [%d] , cfd = [%d]", lfd,cfd);
    int n = 0;
    char buf[1024];
    while(1) {
        memset(buf,0x00,sizeof buf);
        n = read(cfd,buf, sizeof(buf));
        if(n<=0) {
            // 关闭连接，跳出循环
            printf("close, n := %d",n);
            break;
        }
        printf("n=%d , buf =%s\n", n, buf);
        // for(int i=0;i<n;++i) {
        //     buf[i] = toupper(buf[i]);
        // }
        //发送数据
        write(cfd,buf,n);
    }
    close(lfd);
    close(cfd);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/types.h>
 
#include <unistd.h>
#include <arpa/inet.h>
#include <netinet/in.h>


 
int main(void) {
    
    
    int cfd = socket(AF_INET, SOCK_STREAM, 0);
    if(cfd <0 ) {
        perror("socket error");
        exit(1);
    }
    struct sockaddr_in serv;
    bzero(&serv, sizeof (serv));
    serv.sin_family = AF_INET;
    serv.sin_port = htons(8888);
    //serv.sin_addr.s_addr = htonl(INADDR_ANY); //使用本机任意可用IP
    inet_pton(AF_INET, "127.0.0.1",&serv.sin_addr.s_addr);
    //printf("[%p]", serv.sin_addr.s_addr);
    int ret = connect(cfd,(struct sockaddr *)&serv, sizeof serv);
    if (ret <0 ) {
      perror("error^");
      return -1;
    }
    int n=0;
    char buf[256];
    while(1) {
      memset(buf,0,sizeof buf);
      n = read(STDIN_FILENO,buf, sizeof buf);
      printf("input := %s",buf);
      int cmp = strcmp(buf,"quit\n");
      printf("cmp := %d",cmp);
      if(cmp == 0) {
        break;
      }
      //调用网卡设备发送数据
      write(cfd, buf,n);
     // sleep(1000);
      //获取服务端返回
      memset(buf,0,sizeof buf);
      n = read(cfd,buf,sizeof buf);
      if(n<=0) {
        printf("close connection \n");
        break;
      }
      printf(" response := %s",buf);
    }
    close(cfd);

    return 0;
}

注意点

若阻塞函数在阻塞期间被信号收到信号，会被打断

不应该看成错误， errno 此时是EINTR

select 的优点缺点

  1. 代码编写困难
  2. 用户去内核去的轮流拷贝
  3. 最多 1024个客户端连接
    1. 自己修改内核的宏定义
