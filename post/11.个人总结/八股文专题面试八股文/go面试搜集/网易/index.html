#!/bin/bash  
  
log_file="access.log"  # 日志文件路径  
result_file="top_urls.txt"  # 结果文件路径  
  
# 提取URL并计数  
awk -F " " '{print $7}' "$log_file" | sort | uniq -c | sort -nr | head -10 > "$result_file"  
  
echo "脚本执行完成，结果已保存至当前目录的 $result_file"
