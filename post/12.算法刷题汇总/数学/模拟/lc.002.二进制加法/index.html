func reverse(a []int) {
    var n  = len(a) 
    for i:=0;i< n>>1;i++ {
        a[i],a[n-i -1] = a[n-i-1],a[i]
    }
}
func reverse0(a []byte) {
    var n  = len(a) 
    for i:=0;i< n>>1;i++ {
        a[i],a[n-i -1] = a[n-i-1],a[i]
    }
}

func max(a ,b int) int{
    if a > b {
        return a 
    }
    return b
}
func addBinary(a string, b string) string {
    var s1 = make([]int,len(a))
    var s2 = make([]int,len(b))
    for i,v := range a {
        s1[i] = int(v-'0')
    }
    for i,v := range b {
        s2[i] = int(v-'0')
    }

    reverse(s1)
    reverse(s2)
    //var res []int 
   // var rest int 
    var n,m = len(s1),len(s2)
    // 
    s := strings.Builder{}
    var x = 0
    for i:=0;i<max(n,m);i++ {
        if i<n {
            x += s1[i]
        }
        if i<m {
            x += s2[i]
        }
        s.WriteRune( rune((x%2) + '0'))
        x /= 2
    }
    if x > 0 {
        s.WriteRune('1')
    }
    var res []byte
    res  = []byte(s.String())
    reverse0(res)
    return string(res)
}
