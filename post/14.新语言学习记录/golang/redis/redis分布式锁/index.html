package main

import (
	goredislib "github.com/go-redis/redis/v8"
	"github.com/go-redsync/redsync/v4"
	"github.com/go-redsync/redsync/v4/redis/goredis/v8"
)

func main() {
	// Create a pool with go-redis (or redigo) which is the pool redisync will
	// use while communicating with Redis. This can also be any pool that
	// implements the `redis.Pool` interface.
	client := goredislib.NewClient(&goredislib.Options{
		Addr: "localhost:6379",
	})
	pool := goredis.NewPool(client) // or, pool := redigo.NewPool(...)

	// Create an instance of redisync to be used to obtain a mutual exclusion
	// lock.
	rs := redsync.New(pool)

	// Obtain a new mutex by using the same name for all instances wanting the
	// same lock.
	mutexname := "my-global-mutex"
	mutex := rs.NewMutex(mutexname)

	// Obtain a lock for our given mutex. After this is successful, no one else
	// can obtain the same lock (the same mutex name) until we unlock it.
	if err := mutex.Lock(); err != nil {
		panic(err)
	}

	// Do your work that requires the lock.

	// Release the lock so other processes or threads can obtain a lock.
	if ok, err := mutex.Unlock(); !ok || err != nil {
		panic("unlock failed")
	}
}

分布式锁解决方案调研

一个分布式锁需要的基本功能如下：

\1. 互斥性： 只能有一个客户端获得锁

\2. 不会发生死锁： 客户端在持有锁的时候突然 crash掉，最终可以让其他客户端获得锁

\3. 容错性： Redis 集群中大部分的节点正常运行，客户端就可以进行加锁和解锁操作

基于Redis的解决方法

参考 redission框架

https://www.jianshu.com/p/0d3f5395ceaf

基本的功能：

\1. 可重入锁： 获得锁的线程不会锁住自己

\2. 超时时间： 获得锁的进程 如果宕机了， Redis 中存的锁如果不设置过期时间，就会发生死锁，

\3. 看门狗机制, 【如果方法一直在占用锁执行任务， 可以自动延续锁的占用时间】

\4. 红锁算法 【Redis集群下，采用多数原则，大多数节点上锁成功，才能任务是上锁成功】

import (
	"context"
	"errors"
	"github.com/bsm/redislock"
	"github.com/go-redis/redis/v8"
	"log"
	"time"
)

var client = redis.NewClient(&redis.Options{
	Network: "tcp",
	Addr:    "127.0.0.1:6379",
})

func worker(id int) {
	var lock, err = tryForLock(time.Second * 6)
	//fmt.Println(res, err)
	//time.Sleep(time.Second * 100)
	if err != nil {
		log.Println("获得锁失败=== ", err)
		return
	}
	//看门狗，自动续期
	var ctxParent = context.Background()
	var ctx, cancel = context.WithCancel(ctxParent)
	//解锁
	defer func() {
		err = lock.Release(context.Background())
		if err != nil {
			log.Println(" unlock failed !!!", err)
		} else {
			log.Println("===== 解锁成功 ==== id==== ", id, " ====== ")
		}
	}()
	//超时控制
	defer cancel()
	var watchDog = func() {
		for {
			select {
			case <-ctx.Done():
				//被cancel掉了说明任务做完，立刻退出，不要继续加时间了
				return
			default:
				//时间不够了， 再加 4秒时间
				lock.Refresh(ctx, time.Second*4, nil)
			}
		}
	}
	//自动续期
	go watchDog()

	log.Println("获得锁成功, id ======= ", id, "===== doing  task ======")
	time.Sleep(time.Second * 2)
	log.Println("== compliete ===")
}
func main() {
	//3s 无法获得锁，就退出，不抢锁了
	for i := 0; i < 10; i++ {
		go worker(i)
	}
	log.Println(" === ok ====")
	time.Sleep(time.Second * 38)
}

func tryForLock(waitTime time.Duration) (*redislock.Lock, error) {

	// Create a new lock client.
	locker := redislock.New(client)

	ctx := context.Background()
	//var timeout = time.Second * 3
	//3秒 无法获得 就退出
	// Try to obtain lock.
	var timeout = time.After(waitTime)
tryLock:
	for {
		select {
		//case <-ctx.Done():
		case <-timeout:

			log.Println("无法获得锁===")
			return nil, errors.New("超时无法获得锁")
		default:
			//上锁10秒
			ok, err := locker.Obtain(ctx, "mylock", time.Second*8, nil)
			if err == redislock.ErrNotObtained {
				continue tryLock
			} else if err != nil {
				//其他问题
				return nil, err
			}
			return ok, nil

		}

	}

}
