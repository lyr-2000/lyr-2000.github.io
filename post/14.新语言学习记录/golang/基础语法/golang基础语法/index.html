package main

import "fmt"

func f(a float64, op byte, b float64) float64 {
	var res float64
	switch op {
	case '+':
		res = a + b
	case '-':
		res = a - b
	case '*':
		res = a * b
	case '/':
		res = a / b
	}
	return res
}
func main() {
	fmt.Println(f(1, '+', 2))
}
