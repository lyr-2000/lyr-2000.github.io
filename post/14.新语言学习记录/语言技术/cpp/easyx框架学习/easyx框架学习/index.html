#include<graphics.h>
#include <cstdio>
#include<conio.h> 
#include <conio.h>//_getch()的来源头文件，必加！
//#include<stdlib.h>

struct Snake {
	int snake_len;
	//char key;
	int x, y;
};

#define x obj.x
#define y obj.y

#define keyLeft(s)  GetKeyState(72)<0 && s != 'l'

#define keyRight(s)  GetKeyState(76)<0 && s != 'h'
#define keyUp(s)  GetKeyState(75)<0 && s != 'j'
#define keyDown(s)  GetKeyState(74)<0 && s !='k'
#define KeySpace() GetAsyncKeyState(32) 
#define KeyQuit() GetAsyncKeyState(81)
void enter_game() {
	char s[52][52];
	char key = 'l';
	BeginBatchDraw();//
	memset(s, 0, sizeof s);
	int rx = 1 + rand() % 49, ry = 1 + rand() % 49;
	while (s[rx][ry]) {
		rx = 1 + rand() % 49, ry = 1 + rand() % 49;
	}
	s[rx][ry] = -1;
	Snake obj = { 3,25,25 };

	//char tempKey = '0';
	while (1) {
		Sleep(100);
		setfillcolor(WHITE);
		solidrectangle(0, 0, 500, 500);//绘制世界
		//tempKey = _getch();
		if (keyLeft(key)) {
			key = 'h'; //left 
		}
		else if (keyRight(key)) {
			key = 'l'; //right
		}
		else if (keyDown(key)) {
			key = 'j';//down
		} else if (keyUp(key)) {
			key = 'k';//up
		}
		if (KeySpace()) {
			system("pause");
		}
		else if (KeyQuit()) {
			break;
		}
		//printf("key state is %c\n", key);
		if (key == 'h') --x;
		else if (key == 'l') ++x;
		else if (key == 'j') ++y;
		else if (key == 'k') --y;
		if (x <= 0 || y <= 0 || x > 50 || y > 50 || s[x][y] > 0) break; //越界或者吃自己
		s[x][y] = obj.snake_len;
		for (int i = 1; i <= 50; i++) {
			for (int j = 1; j <= 50; j++) {
				if (s[i][j] > 0) {
					s[i][j]--;  //对比：s[x][y] = obj.snake_len;， 经过 n秒后变为0
					setfillcolor(BLACK);
					solidrectangle((i - 1) * 10,
						(j - 1) * 10,
						i * 10,
						j * 10
					);
				} else if (s[i][j] == -1) {
					setfillcolor(RED);
					solidrectangle((i - 1) * 10,
						(j - 1) * 10,
						i * 10,
						j * 10
					);
				}
				if (x == rx && y == ry) {
					obj.snake_len++;
					while (s[rx][ry]) {
						//蛇在这个地方
						rx = 1 + rand() % 49;
						ry = 1 + rand() % 49;
					}
					s[rx][ry] = -1; //食物
				}
			}
		}
		FlushBatchDraw();

		//envent loop end.

	}

	setfillcolor(BLACK);
	solidrectangle(0, 0, 500, 500);
	settextcolor(WHITE);
	settextstyle(20, 0, "楷体");
	char tips[100];
	sprintf_s(tips, "game over,your score %d ,press q to quit!", obj.snake_len-3);
	outtextxy(100, 200,tips );



	FlushBatchDraw();//刷新
	Sleep(4000);
	//return '0';
}

int main(void) {
	 
	HWND hwnd = initgraph(640, 520);
	SetWindowTextA(hwnd, "贪吃蛇");
	while (1) {
		
		enter_game();
		system("pause");
		char p = _getch();

		if (p =='q' || p == 'Q') {
			break;
		}
		printf("type key %c\n", p);
		
	}
	
	closegraph();
	return 0;
	//loadimage();
}
