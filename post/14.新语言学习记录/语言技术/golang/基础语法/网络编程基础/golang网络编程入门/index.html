package main

import (
	"fmt"
	"net"
)

func main() {
	//0.0.0.0 表示本地地址
	f, err := net.Listen("tcp", "0.0.0.0:8080")
	if err != nil {
		fmt.Printf("errInfo := %#v", err)
		return
	}
	defer f.Close()
	for {
		fmt.Println("客户端连接")
		con, err := f.Accept()
		if err != nil {
			fmt.Printf("err:=%#v\n", err)
			continue
		} else {
			//连接成功
			fmt.Printf("%#v\n", con)
		}

	}

}
