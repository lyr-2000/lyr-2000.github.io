package main

import (
	"fmt"
	"net"
)

func proccess(con net.Conn) {
	defer func(con net.Conn) {
		con.Close()
		fmt.Printf("连接退出")
	}(con)
	//关闭连接
	buf := make([]byte, 1024)
	for {
		n, err := con.Read(buf)
		if err != nil {
			fmt.Printf("%#v\n", err)
			break
		}
		fmt.Printf("msg:=%s\n", buf[0:n])

	}

}

func main() {
	//0.0.0.0 表示本地地址
	f, err := net.Listen("tcp", "0.0.0.0:8080")
	if err != nil {
		fmt.Printf("errInfo := %#v", err)
		return
	}
	defer f.Close()
	for {

		con, err := f.Accept()
		if err != nil {
			fmt.Printf("err:=%#v\n", err)
			continue
		} else {
			fmt.Printf("客户端连接, %#v\n", con)
			//连接成功

			go proccess(con)

		}

	}

}
