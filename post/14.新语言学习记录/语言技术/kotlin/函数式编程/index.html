fun onlyIf(isRun:Boolean ,block:()->Unit) {

        block()
    }
}

fun main(args: Array<String>) {
//
    open class Demo {
        init {
            println("init ___")

        }
        //多个构造函数
        constructor(name:String ,age: Int) : super() {
            println("name = $name, age = $age ")
        }
        constructor(age: Int) :super() {
            println("age := $age")
        }

    }

    var demo = Demo("hello",18)
    println(demo)

}
